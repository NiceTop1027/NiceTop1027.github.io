## 순열 재귀 함수 분석하기!

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

void PrintV(vector<int> v) {
    for (int i : v) {
        cout << i << " ";
    }
    cout << '\n';
}

void MakePermutation(vector<int> v, int n, int r, int depth) {
    if (depth == r) {
        PrintV(vector<int>(v.begin(), v.begin() + r));
        return;
    }

    for (int i = depth; i < n; i++) {
        swap(v[i], v[depth]); 
        MakePermutation(v, n, r, depth + 1);   
        swap(v[i], v[depth]);         
    }
}

int main() {
    vector<int> v = {1, 2, 3, 4};
    int n = v.size();
    int r = 2;

    MakePermutation(v, n, r, 0);
    return 0;
}
```
