class IntArrayList {
private:
    int* arr;
    int size;
    int capacity;

    void expand(){
        capacity *= 2;
        int* newArr = new int[capacity];
        for (int i = 0; i < size; i++){
            newArr[i] = arr[i];
        }
        delete[] arr;
        arr = newArr;
    }

public:

        arr = new int[capacity];
    }

    ~IntArrayList(){
        delete[] arr;
    }

        if (size == capacity){

    }

            arr[i] = arr[i - 1];
        }
        arr[index] = value;
        size++;
